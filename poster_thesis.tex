  \begin{alertblock}{Motivation \& Problem Statement}
    Standard GNN pipelines rest on \textbf{two flawed structural assumptions}:
    \begin{itemize}
      \item \textbf{Homogeneity:} Only citation edges used --- co-authorship,
            venue, and field-of-study signals discarded.
      \item \textbf{Undirectedness:} Direction erased, losing the distinction
            between a \emph{seminal paper} (high in-degree) and a
            \emph{survey} (high out-degree).
    \end{itemize}
    Together these cause \textbf{Minority Class Collapse}. \textbf{Class~29
    (cs.NI)} falls below 20\% test accuracy while global accuracy exceeds
    77\% --- 44\% of Networking papers misclassified as NLP.
  \end{alertblock}

  \begin{block}{Dataset: OGBN-arXiv}
    \begin{center}
    \begin{tabular}{lr}
      \toprule
      \textbf{Property} & \textbf{Value}\\
      \midrule
      Nodes $|V|$        & 169{,}343 papers\\
      Edges $|E|$        & 1{,}166{,}243 directed links\\
      Feature dim.       & 128 / \textbf{1524} (Late-Fusion)\\
      Classes $C$        & 40 arXiv CS categories\\
      Train / Val / Test & 90{,}941 / 29{,}799 / 48{,}603\\
      \bottomrule
    \end{tabular}
    \end{center}
    Temporal split: test nodes have sparser context.
    Class imbalance spans \textbf{two orders of magnitude}.

    \smallskip\noindent\textbf{Multigraph subgraph properties:}
    \begin{center}
    \begin{tabular}{lrrc}
      \toprule
      \textbf{Subgraph} & \textbf{$|E|$} & \textbf{Deg.} & \textbf{$h_E$}\\
      \midrule
      References   & 2{,}315{,}598 & 13.7 & 0.654\\
      Authorship   & 6{,}749{,}335 & 39.9 & 0.580\\
      Source/Venue &   605{,}660   &  3.6 & 0.590\\
      Subject Area & 8{,}279{,}687 & 48.9 & \textcolor{negativered}{0.319}\\
      \bottomrule
    \end{tabular}
    \end{center}
    Low subject-area homophily ($h_E{=}0.319$) motivated
    \textbf{homophily-weighted aggregation}.
  \end{block}

  \begin{block}{Theoretical Background}
    Standard MPNNs aggregate only over \emph{incoming} neighbours:
    \[
      a^{(t)}(v)=\operatorname{AGG}\!\bigl(\{h^{(t-1)}(u)\mid u\in\mathcal{N}_\text{in}(v)\}\bigr)
    \]
    \textbf{Thm~1} (Egressy et al., AAAI-24): a standard MPNN
    \emph{cannot} distinguish nodes with identical in-neighbourhoods but
    differing out-degree. Symmetrising does not help.

    \smallskip
    \textbf{Thm~2}: \textbf{Reverse Message Passing} with separate channels
    over $\mathcal{N}_\text{in}$ \emph{and} $\mathcal{N}_\text{out}$
    \emph{provably} solves directed out-degree detection.
  \end{block}

\end{column}

  \begin{block}{Proposed 4-Phase Pipeline}
    \begin{center}
      \pipelinediagram
    \end{center}
  \end{block}

  \begin{block}{Phase 1: Directed Multigraph Message Passing}
    Citation edges \textbf{split into two typed channels}:
    \[
      E_\text{cites}=\{(u,v)\},\quad E_\text{cited\_by}=\{(v,u)\}
    \]
    \texttt{HeteroConv} wraps GraphSAGE with independent weight matrices per
    edge type, weighted by empirical $h_E^{(r)}$.

    \textbf{Result:} Untuned Run~1 hit \textbf{79.02\% validation},
    clearing the 77--78\% undirected plateau immediately.
  \end{block}

  \begin{block}{Phase 2: Targeted Regularisation}
    Hub-node memorisation broken via:
    \begin{itemize}
      \item \textbf{Targeted DropEdge:} $p_\text{cited\_by}{=}0.50$,
            $p_\text{cites}{=}0.20$.
      \item \textbf{Input Perturbation:}
            $\tilde{x}_v{=}x_v{+}\varepsilon_v$,\;
            $\varepsilon_v{\sim}\mathcal{N}(0,0.01^2 I)$ --- soft Ego-ID.
      \item \textbf{3-Model Ensemble:} logit averaging across seeds.
    \end{itemize}
    \textbf{Result:} $+$0.53\% micro-accuracy $\Rightarrow$
    project peak \textbf{78.14\%}.
  \end{block}

  \begin{block}{Phase 3: GraphSMOTE Topology Balancing}
    Louvain: \textbf{$>$50\% of Class~29 training nodes isolated} ---
    45 disconnected sub-communities.

    Synthetic nodes via feature interpolation:
    \[
      \tilde{x}=x_i+\lambda(x_j-x_i),\quad\lambda\sim\mathcal{U}(0,1)
    \]
    MLP edge predictor wires $\tilde{v}$ into $E_\text{cites}$ and
    $E_\text{cited\_by}$.

    \textbf{Result:} \textbf{10{,}090 synthetic nodes}, \textbf{16 classes}.
    Class~12 (5 real nodes) $\to$ \textbf{100\% test accuracy}.
  \end{block}

\end{column}

  \begin{block}{Phase 4: Lexical-Semantic Late-Fusion}
    Dense SimTG embeddings cluster cross-disciplinary papers near
    vocabulary-overlapping classes; heterophilic aggregation
    \textbf{drags representations to the wrong centroid}.

    Concatenate 500-dim TF-IDF flags with 1024-dim SimTG:
    \[
      X_\text{fusion}=\bigl[X^\text{1024}_\text{SimTG}\;\|\;
        X^\text{500}_\text{TF-IDF}\bigr]\in\mathbb{R}^{N\times1524}
    \]
    TF-IDF dims are \textbf{hard lexical switches} immune to heterophilic
    neighbourhood pooling.

    \textbf{Per-class accuracy before vs.\ after Late-Fusion:}
    \begin{center}
    \begin{tabular}{llcc}
      \toprule
      \textbf{Cl.} & \textbf{Category} & \textbf{Before} & \textbf{After}\\
      \midrule
      29 & cs.NI Networking   & $<$20\% & \textcolor{positivegreen}{26.7\%}\\
      11 & cs.AR HW Arch.     & $<$15\% & \textcolor{positivegreen}{18.3\%}\\
      21 & cs.MS Math.\ Soft. & $<$10\% & \textcolor{positivegreen}{14.6\%}\\
      12 & cs.CE Comp.\ Eng.  & 0.0\%   & \textcolor{positivegreen}{\textbf{100\%}}\\
       3 & cs.CG Comp.\ Geom. & $<$25\% & \textcolor{positivegreen}{39.8\%}\\
      \midrule
      \multicolumn{2}{l}{\textbf{Global Macro-Acc.}}
                              & $<$50\% & \textcolor{positivegreen}{\textbf{65.99\%}}\\
      \bottomrule
    \end{tabular}
    \end{center}
  \end{block}

  \begin{block}{Negative Result: Virtual Super-Node Experiment}
    Synthetic hub $v^*_c$ connected to all minority training members.
    At test time no hub exists --- \textbf{Super-Node Addiction}.
    \begin{center}
    \begin{tabular}{lcc}
      \toprule
      \textbf{Config} & \textbf{Macro (\%)} & \textbf{$\Delta$}\\
      \midrule
      Phase~4 baseline & 65.99 & ---\\
      + Super-Nodes    & \textcolor{negativered}{64.16} & \textcolor{negativered}{$-$1.83}\\
      \quad Class~29   & \textcolor{negativered}{14.2}  & \textcolor{negativered}{$-$12.5}\\
      \quad Class~3    & \textcolor{negativered}{10.7}  & \textcolor{negativered}{$-$29.1}\\
      \bottomrule
    \end{tabular}
    \end{center}
    \textbf{Lesson:} feature-level fusion strictly preferable to train-only
    topological injection.
  \end{block}

  \begin{block}{Consolidated Ablation \& Results}
    \begin{center}
    \begin{tabular}{lcc}
      \toprule
      \textbf{Configuration} & \textbf{Micro} & \textbf{Params}\\
      \midrule
      Paper baseline (Ly et al.\ 2024) & 77.21\% & ---\\
      + Directed edges (Ph.~1)         & 77.61\% & 0.8M\\
      + DropEdge + Ens.\ (Ph.~2)       & \textbf{78.14\%} & 1.26M\\
      + GraphSMOTE (Ph.~3)             & 77.98\% & 1.26M\\
      + TF-IDF Fusion (Ph.~4)          & 77.98\% & 1.66M\\
      \midrule
      Super-Nodes (neg.)               & 77.67\% & 1.26M\\
      TransformerConv + Fusion         & 77.67\% & 2.60M\\
      \textbf{GraphSAGE + Fusion}      & \textbf{77.74\%} & \textbf{1.26M}\\
      \bottomrule
    \end{tabular}
    \end{center}
    GraphSAGE (1.26M) beats TransformerConv (2.6M) in accuracy \emph{and}
    speed (55 vs.\ 80 epochs). \textbf{McNemar:}
    $\chi^2{=}28.79$, $p{=}8.07{\times}10^{-8}$.
  \end{block}

  \begin{exampleblock}{Key Contributions}
    \begin{enumerate}
      \item \textbf{Directed Multigraph (Reverse MP):} first AAAI-24
            application to OGBN-arXiv; clears undirected plateau untouched.
      \item \textbf{Late-Fusion:} macro $<$50\%
            $\to$ \textbf{65.99\%} across all 40 categories.
      \item \textbf{GraphSMOTE:} 10{,}090 synthetic nodes;
            Class~12 $\to$ \textbf{100\% accuracy}.
      \item \textbf{Negative result:} topology injection $<$ feature fusion
            for heterophily resolution.
      \item \textbf{Validated:} 78.14\% micro, 65.99\% macro,
            $p{=}8.07{\times}10^{-8}$.
    \end{enumerate}
  \end{exampleblock}
